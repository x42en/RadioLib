139 long pulseIn(uint32_t pin, uint32_t state, unsigned long timeout) override {
140 if (pin == RADIOLIB_NC) {
141 return 0;
142 }
143
144 unsigned long startMicros = micros();
145 unsigned long currentMicros;
146
147 // Wait for pulse to start
148 while (digitalRead(pin) != state) {
149 currentMicros = micros();
150 if (currentMicros - startMicros >= timeout) {
151 return 0;
152 }
153 }
154
155 // Measure pulse duration
156 unsigned long pulseStart = micros();
157 while (digitalRead(pin) == state) {
158 currentMicros = micros();
159 if (currentMicros - pulseStart >= timeout) {
160 return 0;
161 }
162 }
163
164 return micros() - pulseStart;
165 }

209 // SPI operations using spi_master driver
210 void spiBegin() {
211 ESP_LOGD("EspHal", "Initializing SPI on host %d (SCK=%d, MISO=%d, MOSI=%d)",
212 spiHost, spiSCK, spiMISO, spiMOSI);
213
214 // Try to initialize the bus - it may already be initialized by another
215 // component
216 spi_bus_config_t bus_config = {};
217 bus_config.mosi_io_num = spiMOSI;
218 bus_config.miso_io_num = spiMISO;
219 bus_config.sclk_io_num = spiSCK;
220 bus_config.quadwp_io_num = -1;
221 bus_config.quadhd_io_num = -1;
222 bus_config.data4_io_num = -1;
223 bus_config.data5_io_num = -1;
224 bus_config.data6_io_num = -1;
225 bus_config.data7_io_num = -1;
226 bus_config.max_transfer_sz = 4096;
227 bus_config.flags = 0;
228 bus_config.isr_cpu_id = ESP_INTR_CPU_AFFINITY_AUTO;
229 bus_config.intr_flags = 0;
230
231 esp_err_t ret = spi_bus_initialize(spiHost, &bus_config, SPI_DMA_CH_AUTO);
232 if (ret == ESP_OK) {
233 busInitialized = true;
234 ESP_LOGD("EspHal", "SPI bus initialized successfully");
235 } else if (ret == ESP_ERR_INVALID_STATE) {
236 // Bus already initialized - this is OK for shared bus
237 busInitialized = false; // We didn't init it, so we won't free it
238 ESP_LOGD("EspHal", "SPI bus already initialized (shared bus)");
239 } else {
240 ESP_LOGE("EspHal", "Failed to initialize SPI bus: %s",
241 esp_err_to_name(ret));
242 }
243 }

252 uint8_t spiTransferByte(uint8_t b) {
253 if (spiDevice == nullptr) {
254 ESP_LOGE("EspHal", "SPI device not initialized!");
255 return 0xFF;
256 }
257
258 spi_transaction_t trans = {};
259 trans.flags = SPI_TRANS_USE_TXDATA | SPI_TRANS_USE_RXDATA;
260 trans.length = 8; // 8 bits
261 trans.tx_data[0] = b;
262
263 esp_err_t ret = spi_device_polling_transmit(spiDevice, &trans);
264 if (ret != ESP_OK) {
265 ESP_LOGE("EspHal", "SPI transfer failed: %s", esp_err_to_name(ret));
266 return 0xFF;
267 }
268
269 uint8_t received = trans.rx_data[0];
270
271 // Debug log for first few transfers
272 static int transfer_count = 0;
273 if (transfer_count < 20) {
274 ESP_LOGD("EspHal", "SPI: TX=0x%02X RX=0x%02X", b, received);
275 transfer_count++;
276 }
277
278 return received;
279 }

281 void spiTransfer(uint8_t *out, size_t len, uint8_t *in) {
282 if (spiDevice == nullptr) {
283 ESP_LOGE("EspHal", "SPI device not initialized!");
284 return;
285 }
286
287 if (len == 0) {
288 return;
289 }
290
291 spi_transaction_t trans = {};
292 trans.length = len * 8; // length in bits
293 trans.tx_buffer = out;
294 trans.rx_buffer = in;
295
296 esp_err_t ret = spi_device_polling_transmit(spiDevice, &trans);
297 if (ret != ESP_OK) {
298 ESP_LOGE("EspHal", "SPI transfer failed: %s", esp_err_to_name(ret));
299 }
300 }

309 void spiEnd() {
310 // Remove device from bus
311 if (deviceAdded && spiDevice != nullptr) {
312 spi_bus_remove_device(spiDevice);
313 spiDevice = nullptr;
314 deviceAdded = false;
315 ESP_LOGD("EspHal", "SPI device removed");

318 // Free the bus only if we initialized it
319 if (busInitialized) {
320 spi_bus_free(spiHost);
321 busInitialized = false;
322 ESP_LOGD("EspHal", "SPI bus freed");
323 }

332 esp_err_t addSpiDevice(int8_t csPin, uint32_t clockHz = 500000) {
333 if (csPin == RADIOLIB_NC) {
334 ESP_LOGE("EspHal", "Invalid CS pin");
335 return ESP_FAIL;
336 }
337
338 spi_device_interface_config_t dev_config = {};
339 dev_config.command_bits = 0;
340 dev_config.address_bits = 0;
341 dev_config.dummy_bits = 0;
342 dev_config.mode = 0; // SPI mode 0 (CPOL=0, CPHA=0)
343 dev_config.clock_source = SPI_CLK_SRC_DEFAULT;
344 dev_config.duty_cycle_pos = 128; // 50% duty cycle
345 dev_config.cs_ena_pretrans = 0;
346 dev_config.cs_ena_posttrans = 0;
347 dev_config.clock_speed_hz = (int)clockHz;
348 dev_config.input_delay_ns = 0;
349 dev_config.spics_io_num = csPin;
350 dev_config.flags = 0;
351 dev_config.queue_size = 1;
352 dev_config.pre_cb = nullptr;
353 dev_config.post_cb = nullptr;
354
355 esp_err_t ret = spi_bus_add_device(spiHost, &dev_config, &spiDevice);
356 if (ret != ESP_OK) {
357 ESP_LOGE("EspHal", "Failed to add SPI device: %s", esp_err_to_name(ret));
358 return ret;
359 }
360
361 deviceAdded = true;
362 ESP_LOGD("EspHal", "SPI device added (CS=%d, clock=%u Hz)", csPin, clockHz);
363 return ESP_OK;
364 }
365
366 esp_err_t removeSpiDevice() {
367 if (!deviceAdded || spiDevice == nullptr) {
368 ESP_LOGW("EspHal", "No SPI device to remove");
369 return ESP_OK;
370 }
371
372 esp_err_t ret = spi_bus_remove_device(spiDevice);
373 if (ret != ESP_OK) {
374 ESP_LOGE("EspHal", "Failed to remove SPI device: %s",
375 esp_err_to_name(ret));
376 return ret;
377 }
378
379 spiDevice = nullptr;
380 deviceAdded = false;
381 ESP_LOGD("EspHal", "SPI device removed");
382 return ESP_OK;
383 }
